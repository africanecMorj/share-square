Під'єднання mongoDb та node-telegram-bot-api:
	
	створюєте .env файл в корені, додаєте MONGO_URI, TG_URI

api:
	post(`sendUrl`):
	
		body: 
			
			URL: посилання яке потрібно передати;
			login: логін користувача.Якщо користувач в гостьовому режимі вказуєте значення false;
			password: пароль користувача, для гостьового режиму значення аналогічне;
		
		можливі відповіді:
			
			`You have exceeded the limit. Await 5 minute`: з одного айпі можливо надсилати лише один запит в п'ять хвлини.Почекайте п'ять хвилин і повторіть запит;
			`Incorrect url`: значення яке було надіслене в параметрі URL не є посиланням.Вкажіть вірне посилання і повторіть спробу;
			`Too long url`:значення яке було надіслане в параметрі URL перевищило 170 символів. Вкажіть посилання менше 170 символів і повторіть спробу;
			`This login doesn't exist`: вказаного логіну в параметрі login немає в базі данних. Вкажіть вірний логін і повторіть спробу;
			`Incorrect password`:вказаного пароля в параметрі password немає в базі данних.Вкажіть вірний пароль і повторіть спробу;
			п'ятизначне число: посилання додалось до бази данних і буде видалено через сім днів Це число є кодом доступу до посилання;
	
	get(`getUrl/:code`):
		
		params:
			
			code:код доступу до посилання

		можливі відповіді:
			
			`This code doesn't exist`: код доступу не вірний;
			ваше посилання: запит успішно відбувся;
	
	delete(`/deleteUrl/:code/:login/:password`):
		
		params:

			code: код доступу до посилання яке потрібно видалити;
			login: логін користувача в аккаунті якого зберігалось посилання, для гостьового режиму вказуєте значення false
			password: пароль користувача в аккаунті якого збераігалось посилання, для гостьового режиму значення аналогінче
			
		можливі відповіді:
			
			`This code doesn't exist`: коду який ви надіслали в параметрі code немає в базі данних. Відповідь приходе якщо значення параметру login - false;
			`This code or login doesn't exist`: вказаного логіну або коду немає в базі данних;
			`Incorrect password`: вказаного пароля немає в базі данних;
			`Succesfully deleted`: посилання успішно видалено;
	
	post(`/registration`):

		body:
			
			password: пароль аккаунту який потрібно зареєструвати;
			login: логін аккаунту який потрібно зареєструвати;
		
		можливі відповіді:
			
			`Succesfully regisrated`: аккаунт успішно додано в систему;
			`Login is occupied`: логін зайнято. Виберіть інишй логін і повторіть спробу;
	
	get(`/signIn/:password/:login`):
		
		params:
			
			password: пароль аккаунту з якого потрібно увійти в систему;
			login: логін аккаунту з якого потрбно увійти в систему;
		
		моливі відповіді:
		
			`Incorrect password`: пароля немає в базі данних;
			`Incorrect login`: логіна немає в базі данних;
			{mesage:`Succesfully signed in`, history:array}: ви успішно увійшли і дістали історію своїх посилань;
